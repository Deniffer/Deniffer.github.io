// from objc-private截取 为什么多了这个东西出来，噢噢是objc_class要inheritance的东西
struct objc_object {
private:
    isa_t isa;
public:
    // ISA() assumes this is NOT a tagged pointer object
    Class ISA(bool authenticated = false);

    // rawISA() assumes this is NOT a tagged pointer object or a non pointer ISA
    Class rawISA();

    // getIsa() allows this to be a tagged pointer object
    Class getIsa();
    
    uintptr_t isaBits() const;

    // initIsa() should be used to init the isa of new objects only.
    // If this object already has an isa, use changeIsa() for correctness.
    // initInstanceIsa(): objects with no custom RR/AWZ
    // initClassIsa(): class objects
    // initProtocolIsa(): protocol objects
    // initIsa(): other objects
    void initIsa(Class cls /*nonpointer=false*/);
    void initClassIsa(Class cls /*nonpointer=maybe*/);
    void initProtocolIsa(Class cls /*nonpointer=maybe*/);
    void initInstanceIsa(Class cls, bool hasCxxDtor);
   //..... 很多其他的东西
}

// from objc-runtime-new  截取
struct objc_class : objc_object {
  // 这四个CTOR 其实可以不用关心 
  objc_class(const objc_class&) = delete;
  objc_class(objc_class&&) = delete;
  void operator=(const objc_class&) = delete;
  void operator=(objc_class&&) = delete;
    // Class ISA;
    Class superclass;         // pointer towards superclass
    cache_t cache;             // formerly cache pointer and vtable
    class_data_bits_t bits;    // class_rw_t * plus custom rr/alloc flags
    // ....  很多其他的东西
}

从上述的code可以很容易地看出 object_class （public inheritance）于 objc_object，于是我们得出结论，每一个objc_class都是一个objc_object 。所有的object（实例） 到最后都是一个structure objc_object（这就是事实）。而所谓的class 其实只不过是在 object的基础上加了一点很多东西。

如果直接开始讲isa_t 这个union多少会有点突兀， 所以我们先来看看简单的做法（也就是已经过时的OC1.0）即下图的左半部分

上图可能不能够明显指出OC1.0 和2.0 的区别与联系，那我们来看看下面这张图

由于篇幅以及OC1.0已经obsolete的关系，(花了一点时间后)发现研究OC1.0 已经是一件没有意义的事情 ，可以看到从1.0到2.0 ，objc_class 很多的instance variable 被封装成 struct class_data_bits的成员，以及多了一个union isa_t。其他则原封不动的（如cache 和supe_class）迁移到 OC 2.0。 但是研究OC 1.0 -> OC 2.0的演进还是一件很有意义的事情。（由于时间关系 下次一定）

在OC2.0,当一个对象的实例方法被调用的时候，会通过isa找到相应的类，然后在该类的class_data_bits_t中去查找方法。 struct class_data_bits_t是指向了类对象的数据区域。在该数据区域内查找相应方法的对应实现。

但是在我们调用类方法的时候，类对象的isa里面是什么呢？这里为了和对象查找方法的机制一致，遂引入了元类(meta-class)的概念。(位于class_ro_t 里面的metaClass 可以访问public interface )

在引入元类之后，类对象和对象查找方法的机制就完全统一了。

  • 对象的实例方法调用时，通过对象的 isa 在类中获取方法的实现。

  • 类对象的类方法调用时，通过类的 isa 在元类中获取方法的实现。

  • meta-class之所以重要，是因为它存储着一个类的所有类方法（class method Not Instance method）。每个类都会有一个单独的meta-class。

  • 下图很好的解释了isa的指向和superclass 。

    • Root class (class)其实就是NSObject，NSObject是没有超类的，所以Root class(class)的superclass指向nil。
    • 每个Class都有一个isa指针指向唯一的Meta class
    • Root class(meta)的superclass指向Root class(class)，也就是NSObject，形成一个回路。
    • 每个Meta class的isa指针都指向Root class (meta)。
    • 类对象和元类对象是唯一的，对象（instance）是可以在运行时创建无数个的。

struct class_data_bits_t {
    friend objc_class;
    // Values are the FAST_ flags above.
    uintptr_t bits;
private:
    // GETTER 
    bool getBit(uintptr_t bit) const
    {
        return bits & bit;
    }
     // 一些类似的getter/setter方法
public:
    // important getter 如果对struct class_rw_t 感兴趣可以看看源码，由于篇幅关系
    // 不再展示class_rw_t 的源码 ，可以看下图的内存布局进行感性认识
    class_rw_t* data() const {
        // 与fast_data_mask（其实就是一个地址） 做与操作得到数据地址
        return (class_rw_t *)(bits & FAST_DATA_MASK); 
    }
    void setData(class_rw_t *newData)
    {
        ASSERT(!data()  ||  (newData->flags & (RW_REALIZING | RW_FUTURE)));
        // Set during realization or construction only. No locking needed.
        // Use a store-release fence because there may be concurrent
        // readers of data and data's contents.
        uintptr_t newBits = (bits & ~FAST_DATA_MASK) | (uintptr_t)newData;
        atomic_thread_fence(memory_order_release);
        bits = newBits;
    }

    // Get the class's ro data, even in the presence of concurrent realization.
    // fixme this isn't really safe without a compiler barrier at least
    // and probably a memory barrier when realizeClass changes the data field
    const class_ro_t *safe_ro() const {
        class_rw_t *maybe_rw = data();
        if (maybe_rw->flags & RW_REALIZED) {
            // maybe_rw is rw
            return maybe_rw->ro();
        } else {
            // maybe_rw is actually ro
            return (class_ro_t *)maybe_rw;
        }
    }

图示为structure class_data_bits_t bits的持有关系(建议看看源码是怎么合理划分访问权限)

bits （实际上就是data的地址 不过加上了一些flag，注释上有说明）指向类的数据区域，里面有很多好东西，包括但不限于method_array,property_array,protocol_array，

在编译期,类的结构中的 class_data_bits_t *data指向的是一个 class_ro_t *指针, 在运行时(runtime)调用 realizeClass方法，将 class_rw_t casting为 class_ro_t指针 ,强烈建议看看

static Class realizeClassWithoutSwift(Class cls, Class previously)

这个method不仅初始化了superClass/metaclass ，而且把类里面的属性，协议，方法都加载进来（通过调用methodizeClass）
