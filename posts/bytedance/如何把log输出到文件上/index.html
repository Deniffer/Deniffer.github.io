明确诉求， 将每次log都打到一个文件上，实现的功能点

  • 以每天每机型作为一个文件的生成时间

  • 不同的调试信息 有不同的优先级(不同的颜色) 每条log应该要带上时间

  • 文件名 mm-dd-iphonexx.log

  • 应该很容易就能和NSLog作替换

  • 真机调试 不能写在mac上

实现方案:

尽可能的mini，后期添加功能， 明确输入位置~/workspace/logs/fileName ,

函数名 IESLiveLog(@" “) 重名 更改为DIZLog or XXLog

Simulator

  • 直接写入本地 NSFileManager defaultManager

真机调试

  • Smb 服务共享文件 Objective-C无法找到 apple对于应用 文件之间是强隔离的

  • 做文件的sync操作，也是走的共享文件到mac的思想。费劲且效果不好。

  • 做C/S架构

    • 真机做webserver 服务暴露接口给mac 费劲且消耗性能
      • server内嵌入aweme 改动量大，需依赖第三方库
      • 另起一个app做webserver ，app进入后台最多十分钟之后就会挂起，无法持续有效的服务
    • mac端做server，真机request ，mac收到request之后写入

void liveLog(NSString *format, ...)
  //NS_FORMAT_FUNCTION(1, 2)
    // The NS_FORMAT_FUNCTION attribute tells the compiler to treat the 1st argument like
    // a format string, with values starting from the 2nd argument. This way, you'll
    // get the proper warnings if format specifiers and arguments don't match.
{
    va_list args;
    va_start(args, format);

    // get current Date & Time
    NSDate* now = [NSDate date];
    NSDateFormatter *formatter = [[NSDateFormatter alloc] init];
    [formatter setDateFormat:@"HH:mm:ss"];
    NSString *currentTime = [formatter stringFromDate:now];
    [formatter setDateFormat:@"MM-dd,HH"];
    NSString *currentDate = [formatter stringFromDate:now];
    // define msg & filelocation
    NSString *msg = [[NSString alloc] initWithFormat:[NSString stringWithFormat:@"%@",format] arguments:args];
    // parse msg
    NSString *msgFlag = [msg substringWithRange:NSMakeRange(7, 4)];
    NSString *infoLevel = @"\e[0m";
    if ([msgFlag isEqualToString:@"info"]) {
        infoLevel = @"\e[1;33m";
    } else if ([msgFlag isEqualToString:@"warn"]) {
        infoLevel = @"\e[1;34m";
    } else if ([msgFlag isEqualToString:@"dbug"]){
        infoLevel = @"\e[31;35m";
    }
    NSString *writeStr = [NSString stringWithFormat:@"%@[%@][%@] :%@\e[0m\n",infoLevel,currentDate,currentTime,msg];
    NSString *fileLocation = [NSString stringWithFormat:@"/Users/bytedance/workspace/logs/%@.log",currentDate];
    if (![[NSFileManager defaultManager] fileExistsAtPath:fileLocation]){
        [writeStr writeToFile:fileLocation atomically:YES encoding:NSUTF8StringEncoding error:nil];
    } else {
        NSFileHandle *fileHandle = [NSFileHandle fileHandleForWritingAtPath:fileLocation];
        unsigned long long endOfFileOffset;
        if (@available(iOS 13.0, *)) {
            BOOL getOffsetSuccess = [fileHandle seekToEndReturningOffset:&endOfFileOffset error:nil];
            if (getOffsetSuccess) {
                [fileHandle writeData:[writeStr dataUsingEncoding:NSUTF8StringEncoding] error:nil];
            }
        } else {
            // Fallback on earlier versions
        }
    }
    NSDictionary *jsonDict = @{
        @"info_level" : @"",
        @"msg" : @"",
        @"source" : @""
    };
    NSData *data = [NSJSONSerialization dataWithJSONObject:jsonDict options:0 error:nil];
    createUploadTask(data);
    NSLog(@"%@", writeStr);
    va_end(args);
}

#define DIZLogInfo(frmt, ...) do{ LogWrapper(0,(frmt), ##__VA_ARGS__); } while(0)
#define DIZLogWarn(frmt, ...) do{ LogWrapper(1,(frmt), ##__VA_ARGS__); } while(0)
#define DIZLogDebug(frmt, ...) do{ LogWrapper(2,(frmt), ##__VA_ARGS__); } while(0)
#define DIZLogError(frmt, ...) do{ LogWrapper(3,(frmt), ##__VA_ARGS__); } while(0)

void createUploadTask(NSData *data) {
    NSURL *URL = [NSURL URLWithString:@"http://10.95.63.121:10000"];
    NSMutableURLRequest *requset = [[NSMutableURLRequest alloc] initWithURL:URL];
    [requset setHTTPMethod:@"POST"];
    NSURLSession *session = [NSURLSession sessionWithConfiguration:[NSURLSessionConfiguration defaultSessionConfiguration]];
    NSURLSessionUploadTask *uploadTask = [session uploadTaskWithRequest:requset fromData:data];
    [uploadTask resume];
}

void LogWrapper(int log_level,NSString *format, ...) {
    va_list args;
    va_start(args, format);

    // get current Date & Time
    NSDate* now = [NSDate date];
    NSDateFormatter *formatter = [[NSDateFormatter alloc] init];
    [formatter setDateFormat:@"HH:mm:ss"];
    NSString *currentTime = [formatter stringFromDate:now];
    [formatter setDateFormat:@"MM-dd"];
    NSString *currentDate = [formatter stringFromDate:now];
    // define msg & filelocation
    NSString *msg = [[NSString alloc] initWithFormat:[NSString stringWithFormat:@"%@",format] arguments:args];
    NSString *deviceName = [[UIDevice currentDevice] name] ?: @"default";
    NSString *writeStr = [NSString stringWithFormat:@"[%@][%@][%@] :%@",deviceName,currentDate,currentTime,msg];
    NSDictionary *jsonDict = @{
        @"log_level" : @(log_level),
        @"msg" : writeStr,
        @"source" : deviceName
    };
    NSData *data = [NSJSONSerialization dataWithJSONObject:jsonDict options:0 error:nil];
    createUploadTask(data);
    NSLog(@"%@", writeStr);
    va_end(args);
}

from http.server import HTTPServer, BaseHTTPRequestHandler
from datetime import date
import json
 
data = {'result': 'HTTP SERVER OK'}
host = ('10.95.15.149', 9001)
 
class My_Server(BaseHTTPRequestHandler):

    def do_GET(self):
        self.send_response(200)
 
        # 发给请求客户端的响应数据
        self.send_header('Content-type', 'application/json')
        self.end_headers()
        self.wfile.write(json.dumps(data).encode())
 
    def do_POST(self):
 
        datas = self.rfile.read(int(self.headers['content-length']))
        json_data = json.loads(datas.decode())
        self.parse_data(json_data)
    
    def get_current_date(self):
        today = date.today()
        return today.strftime("%m-%d-")
    def parse_data(self,data):
        log_dir = "/Users/bytedance/workspace/logs/"
        color_array = ["\033[1;33m","\033[1;34m","\033[31;35m",""] #\033[1;93m
        file_name = log_dir+ self.get_current_date() + data["source"] + ".log"
        with open(file=file_name , mode="a") as file:
            print(color_array[data["log_level"]] + data['msg'] + "\033[0m \n")
            file.write(color_array[data["log_level"]] + data['msg'] + "\033[0m \n")
            file.close()
 
if __name__ == '__main__':
    server = HTTPServer(host, My_Server)
    print("server启动@ : %s:%s" % host)
 
    server.serve_forever()

Pros & Cons

  • 可以方便快捷的利用双屏优势 看log再也不需要上下拉xcode了

  • 可以把自己的log带到测试包， qa在反馈bug的时候 也能通过log查看 bug具体发生的表现，甚至可以根据log来判断 qa是不是用的最新包

  • Log 可以保存到某个文件

  • 可以同时看N台手机的Log， 方便测试观众和多个主播端的表现

  • 非固定内网ip 一旦ip发生变化就收不到log

  • 需要mac 起后台服务 监听client端的变化

  • 依赖于局域网服务

  • 无法根据log等级过滤
