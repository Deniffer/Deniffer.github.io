CXX = g++
CXXFLAGS= -Wall -g -c -std=c++11
LFLAGS = -lpthread -lrt 
TARGET = mizzo
OBJ = belt.o producer.o consumer.o main.o
IO = io.o

$(TARGET): $(OBJ) $(IO)   # mizzo : belt.o producer.o consumer.o main.o io.o
	$(CXX) -g -o $(TARGET) $(OBJ) $(IO)  $(LFLAGS) # g++ -g -o mizzo belt.o producer.o consumer.o main.o io.o -lpthread -lrt 

$(OBJ): %.o : %.cpp
	$(CXX) $(CXXFLAGS) $< -o $@     # g++ -Wall -g -c -std=c++11 belt.cpp -o belt.o

# add io.c
$(IO) : %.o : %.c
	$(CXX) $(CXXFLAGS) $< -o  $@

.PHONY: clean run depend

depend:
	$(CXX) -MM $(CXXFLAGS) *.cpp >.depend.mk

-include .depend.mk

run:
	./$(TARGET)

clean:
	-rm $(TARGET) *.o
