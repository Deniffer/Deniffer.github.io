#define PGSIZE 4096 // bytes per page
#define PGSHIFT 12  // bits of offset within a page

#define PGROUNDUP(sz)  (((sz)+PGSIZE-1) & ~(PGSIZE-1)) 
#define PGROUNDDOWN(a) (((a)) & ~(PGSIZE-1))

#define PTE_V (1L << 0) // valid
#define PTE_R (1L << 1)
#define PTE_W (1L << 2)  //PTE_R | PTE_W means WRDONLY ! 既可以读也可以写
#define PTE_X (1L << 3)  // X->executed! 
#define PTE_U (1L << 4) //  1 -> user can access

// shift a physical address to the right place for a PTE.
#define PA2PTE(pa) ((((uint64)pa) >> 12) << 10) 

#define PTE2PA(pte) (((pte) >> 10) << 12)

#define PTE_FLAGS(pte) ((pte) & 0x3FF)

// extract the three 9-bit page table indices from a virtual address.
#define PXMASK          0x1FF // 9 bits each level is 9bit(three level)
#define PXSHIFT(level)  (PGSHIFT+(9*(level))) //PGSHIFT = 12  + 9*level
#define PX(level, va) ((((uint64) (va)) >> PXSHIFT(level)) & PXMASK) //get indices of vm

// one beyond the highest possible virtual address.
// MAXVA is actually one bit less than the max allowed by
// Sv39, to avoid having to sign-extend virtual addresses
// that have the high bit set.
#define MAXVA (1L << (9 + 9 + 9 + 12 - 1))  //MAX va

typedef uint64 pte_t;
typedef uint64 *pagetable_t; // 512 PTEs

  • 接下里阅读memlayout.h这个file

    • 需要理解的是在0x0之后和0x80000000之前的 指令都存放在boot rom里面 做一些初始化的指令
    • 然后boot rom 在machine mode 下jump 到0x80000000这个地址，开始映射，可以在entry.S中看到为什么是0x80000000

  • 在接下来阅读vm.c这个file

    • void kvminit() 这个func很明确的看注释就知道他的意思了，创建直接的物理->虚拟的映射，这里的映射是kernel的映射要注意的是。也就是在这个部分，虚拟地址和物理地址是一模一样的。 全称是kernel virtual memory init.这里的每一个部分都对应上memlayout.h的kernel部分。
    • 上面这个func最主要的还是调用了kvmmap，其实就是mappages这个func，来看看这个function的函数实现
